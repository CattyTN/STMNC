#!/usr/bin/env python3
import ujson as json, time, os, io
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler
from pymongo import MongoClient, InsertOne
from datetime import datetime

EVE = "/var/log/suricata/eve.json"
MONGO_URI = os.getenv("MONGO_URI","mongodb://127.0.0.1:27017")
DB="stmnc"; COL="ram"  # đổi nếu bạn dùng DB khác

cli = MongoClient(MONGO_URI)
col = cli[DB][COL]

def map_doc(ev):
    ts = ev.get("timestamp") or ev.get("@timestamp") or datetime.utcnow().isoformat()
    et = ev.get("event_type","flow")
    src = ev.get("src_ip"); dst = ev.get("dest_ip")
    sev = (ev.get("alert") or {}).get("severity")
    sig = (ev.get("alert") or {}).get("signature")
    sport = ev.get("src_port"); dport = ev.get("dest_port"); proto = ev.get("proto")
    desc = f"{src}:{sport} -> {dst}:{dport} {proto} {sig or et}".strip()
    return {
        "TIME_RECEIVE": ts.replace("T"," ")[:19],
        "IP": src or "Chua xac dinh",
        "EXTRACTED_IP": dst or "Chua xac dinh",
        "ALERT_TYPE": et,
        "ALERT_LEVEL_ID": str(sev) if sev is not None else "0",
        "DESCRIPTION": desc,
        "MAC": "Chưa xác định",
        "UNIT_NAME": "Chưa xác định",
        "UNIT_FULL_NAME": "Chưa xác định",
        "USER_NAME": "Chua dinh danh",
        "LABEL": 0
    }

class Tail:
    def __init__(self, path): self.path=path; self.fp=None
    def open(self):
        self.fp = open(self.path,'r',buffering=1,encoding='utf-8',errors='ignore')
        self.fp.seek(0,io.SEEK_END)
    def follow(self):
        while True:
            line=self.fp.readline()
            if not line:
                time.sleep(0.1); continue
            yield line

class RotHandler(FileSystemEventHandler):
    def __init__(self, tail): self.tail=tail
    def on_moved(self,e):
        if e.src_path==EVE: self.tail.open()
    def on_created(self,e):
        if e.src_path==EVE or e.src_path.endswith("eve.json"): self.tail.open()

def main():
    tail=Tail(EVE); tail.open()
    obs=Observer(); obs.schedule(RotHandler(tail),os.path.dirname(EVE) or ".",recursive=False); obs.start()
    buf=[]; BATCH=200; last=time.time()
    try:
        for line in tail.follow():
            line=line.strip()
            if not line: continue
            try:
                obj=json.loads(line)
                events=obj if isinstance(obj,list) else [obj]
                for ev in events:
				    print(json.dumps(ev, ensure_ascii=False))
                    buf.append(InsertOne(map_doc(ev)))
                if len(buf)>=BATCH or time.time()-last>2:
                    col.bulk_write(buf,ordered=False); buf=[]; last=time.time()
            except Exception: pass
    finally:
        if buf: col.bulk_write(buf,ordered=False)
        obs.stop(); obs.join()

if __name__=="__main__": main()

- Báo cáo tiến độ
II. Thành phần và cơ chế hoạt động

Trên máy Client (LAN A)

Khởi chạy chương trình thu thập gói tin viết bằng ngôn ngữ Rust.

Chức năng: bắt gói tin ở tầng datalink, trích xuất src_ip, dst_ip, src_mac, ethertype, sau đó lọc bỏ toàn bộ các kết nối nội bộ.

Chỉ giữ lại các gói có đích nằm ngoài không gian mạng nội bộ (Internet), tức là các kết nối tiềm ẩn nguy cơ rò rỉ hoặc bất thường.

Sau khi lọc, dữ liệu được gửi qua giao thức HTTP POST đến máy chủ Core.

Lý do chọn Rust:

Hiệu năng cao, sử dụng tài nguyên thấp, có thể xử lý lưu lượng lớn theo thời gian thực.

Kiểm soát bộ nhớ an toàn, không lỗi tràn vùng nhớ (buffer overflow) như C/C++.

Dễ đóng gói thành file thực thi độc lập, triển khai nhanh trên nhiều node.
→ Đảm bảo tốc độ thu thập và độ ổn định cao trong điều kiện mạng quân sự giới hạn tài nguyên.

Trên máy Server (LAN B)

Cài đặt API tiếp nhận viết bằng Python Flask.

Endpoint /ingest lắng nghe các yêu cầu HTTP POST từ các client.

Sau khi nhận log, API chuyển đổi và chuẩn hóa thông tin, tra cứu thông tin thiết bị (MAC, đơn vị, người dùng) trong cơ sở dữ liệu MongoDB, sau đó ghi vào collection ram để phân tích.

Lý do chọn Python Flask:

Gọn nhẹ, dễ triển khai, tích hợp trực tiếp với MongoDB và các thư viện phân tích.

Cho phép in log debug, giám sát từng bản ghi nhận được, phù hợp giai đoạn thử nghiệm và diễn tập.

Dễ mở rộng thành microservice độc lập khi cần tích hợp với hệ thống SIEM hoặc OpenCTI.
